#### purpose: Syncs all physical files and database from production site to the dev site ###########
#### module version: MySQL 8.0.x + WordPress 6.4.x #################################################
#### sourced by: n/a ###############################################################################
#### bash aliases: ss sync dev|development #########################################################

## SS-SYNC-DEVELOPMENT DUPLICATES THE DATABASE AND ALL FILES FROM PRODUCTION TO DEV ##
## DEV SITES ARE CONSIDERED INDEPENDENT THUS AUTO-SYNC IS DISABLED BY DEFAULT ##

####################################################################################################
#### TABLE OF CONTENTS (SS-Sync-Development) #######################################################
####################################################################################################

## this is a brief summary of the different code snippets you will find in this script ##
## each section should be commented so you understand what is being accomplished ##

## A. Source SS-Config + SS-Functions
## B. Change To Default Directory
## C. Touch Timestamp File
## D. Message (Begin Script)
## E. Begin If
## F. Dump Production Database
## G. Install Development Database (Conditional)
## H. Import Production Database To Development Database
## I. Copy Relevant Files From Production To Development
## J. Run SS-Install-WordPress-Config (Conditional)
## K. Run SS-Install-WordPress-MU-Plugins (Conditional)
## L. Fix Any Hardcoded Links In Development Theme Files
## M. Fix Incorrect URLs In Development Database
## N. Reset Permissions (WordPress Packages)
## O. End If

####################################################################################################
#### A. SS-Sync-Development: Source SS-Config + SS-Functions #######################################
####################################################################################################

## source ss-config ##
source /var/www/ss-config

## source ss-functions ##
source /var/www/ss-functions

## BELOW THIS RELIES ON SS-CONFIG AND SS-FUNCTIONS

####################################################################################################
#### B. SS-Sync-Development: Change To Default Directory ###########################################
####################################################################################################

## this tells the current script to change to the /var/www directory before proceeding ##
## it can help avoid random warnings and issues when running ss shortcut commands ##

ss_cd_default

####################################################################################################
#### C. SS-Sync-Development: Touch Timestamp File ##################################################
####################################################################################################

## this is a dummy timestamp file that will remember the last time this script was run ##
## it can be useful for developer reference and is sometimes used by SlickStack ##

ss_touch "${TIMESTAMP_SS_SYNC_DEVELOPMENT}"

####################################################################################################
#### D. SS-Sync-Development: Message (Begin Script) ################################################
####################################################################################################

## this is a simple message that announces to the shell the purpose of this bash script ##
## it will only be seen by sudo users who manually run this script in the shell ##

ss_echo "${COLOR_INFO}Running ss-sync-development... ${COLOR_RESET}"

####################################################################################################
#### Query Settings
####################################################################################################

## EXPAND SS DASHBOARD USING MYSQL???
## check if SlickStack dashboard staging.wp_options has paused this script before proceeding

# select EmployeeName from DemoTable where isMarried=true;
# ss_options_staging_sync_status=$(ss_mysql_user -BNe "USE staging; SELECT EmployeeName FROM DemoTable WHERE id=1")
# $sql = "SELECT 'id' FROM Users WHERE username='$name'";

####################################################################################################
#### E. SS-Sync-Development: Begin If ##############################################################
####################################################################################################

if [[ "${DEV_SITE}" == "true" ]] && [[ "${SS_SYNC_DEVELOPMENT}" == "true" ]] && [[ "${WP_MULTISITE}" != "true" ]] && [[ "${SS_LOCKDOWN}" != "true" ]]; then

####################################################################################################
#### F. SS-Sync-Development: Dump Production Database ##############################################
####################################################################################################

## we run ss-dump-database now to export the latest version of production database ##
## this ensures the dump is recent before attempting to sync it to staging ##

source "${PATH_SS_DUMP_DATABASE}"

####################################################################################################
#### G. SS-Sync-Development: Install Development Database (Conditional) ############################
####################################################################################################

## here we perform a simple check to see if the staging database exists yet or not ##
## if the database is not detected the script will attempt to install it now ##

if ! ss_mysql_user -e 'use development'; then
    ss_echo "${COLOR_WARN}no development database found so will try to reinstall it now... ${COLOR_RESET}"
    source "${PATH_SS_INSTALL_MYSQL_DATABASE}"
fi

####################################################################################################
#### H. SS-Sync-Development: Import Production Database To Development Database ####################
####################################################################################################

## this snippet duplicates the dumped production database and imports it into MySQL ##
## then it forecfully replaces the home/siteurl options to staging subdomain ##

## import database ##
ss_cp /var/www/backups/mysql/production.sql /tmp/sync-development.sql
chown mysql:mysql /tmp/sync-development.sql ## does not matter if -L flag used below
chmod 0644 /tmp/sync-development.sql ## does not matter if -L flag used below
ss_mysql_user development < /tmp/sync-development.sql
    
## fix development home and siteurl fields (no mass search/replace to avoid potential conflicts) ##
if [[ "${SS_APP}" == "wordpress" ]]; then

    if [[ -n $(ss_mysql_user --execute "SHOW TABLES FROM development WHERE Tables_in_development LIKE '${DB_PREFIX}options';" 2> /dev/null) ]]; then
        ss_mysql_user --execute="UPDATE development.${DB_PREFIX}options SET option_value='https://dev.${SITE_DOMAIN_EXCLUDING_WWW}' WHERE option_name='home'";
        ss_mysql_user --execute="UPDATE development.${DB_PREFIX}options SET option_value='https://dev.${SITE_DOMAIN_EXCLUDING_WWW}' WHERE option_name='siteurl'";
    fi

fi
    
## delete tmp files ##
ss_rm /tmp/sync-development.sql
    
####################################################################################################
#### I. SS-Sync-Development: Copy Relevant Files From Production To Development ####################
####################################################################################################

## here we copy over everything from production to dev including media uploads folder ##
## very large files e.g. backup archives are always skipped in this snippet ##

## copy wp-content files over (does not skip uploads) ##
ss_rsync --max-size=25mb --exclude 'blacklist.txt' --exclude 'object-cache.php' --exclude='*/mu-plugins/' --exclude='*/temp/' --exclude='*/upgrade/' /var/www/html/wp-content/ /var/www/html/dev/wp-content

## delete potential run-on subdirectories ##
ss_rm /var/www/html/dev/dev*
ss_rm /var/www/html/dev/wp-content/wp-content*

## remove object-cache.php if exists ##
ss_rm /var/www/html/dev/wp-content/object-cache.php*
    
####################################################################################################
#### J. SS-Sync-Development: Run SS-Install-WordPress-Config (Conditional) #########################
####################################################################################################

## below snippet will install wp-config.php in case does not exist already in development ##
## this is a failsafe in case development enabled in ss-config post-installation ##

if [[ ! -f "/var/www/html/dev/wp-config.php" ]]; then
    source "${PATH_SS_INSTALL_WORDPRESS_CONFIG}"
fi

####################################################################################################
#### K. SS-Sync-Development: Run SS-Install-WordPress-MU-Plugins (Conditional) #####################
####################################################################################################

## below snippet will install MU plugins in case they do not exist already in development ##
## this is a failsafe in case development enabled in ss-config post-installation ##

if [[ ! -f "/var/www/html/dev/wp-content/mu-plugins/000-common.php" ]]; then
    source "${PATH_SS_INSTALL_WORDPRESS_MU_PLUGINS}"
fi

####################################################################################################
#### L. SS-Sync-Development: Fix Any Hardcoded Links In Development Theme Files ####################
####################################################################################################

## many themes (or custom-coded themes) often contain hardcoded HTML assets and links ##
## here we forcefully fix these links to ensure development site renders properly ##

find /var/www/html/dev/wp-content/themes/ -type f -exec ss_sed "s#https://${SITE_DOMAIN_INCLUDING_WWW}#https://dev.${SITE_DOMAIN_EXCLUDING_WWW}#g" {} +
find /var/www/html/dev/wp-content/themes/ -type f -exec ss_sed "s#http://${SITE_DOMAIN_INCLUDING_WWW}#https://dev.${SITE_DOMAIN_EXCLUDING_WWW}#g" {} +
find /var/www/html/dev/wp-content/themes/ -type f -exec ss_sed "s#https://${SITE_DOMAIN_EXCLUDING_WWW}#https://dev.${SITE_DOMAIN_EXCLUDING_WWW}#g" {} +
find /var/www/html/dev/wp-content/themes/ -type f -exec ss_sed "s#http://${SITE_DOMAIN_EXCLUDING_WWW}#https://dev.${SITE_DOMAIN_EXCLUDING_WWW}#g" {} +

####################################################################################################
#### M. SS-Sync-Development: Fix Incorrect URLs In Development Database ############################
####################################################################################################

## here we use WP-CLI to search and replace prod domain with development domain in MySQL ##
## this is not super necessary due to wp-config.php but can help avoid issues ##

/usr/local/bin/wp --path=/var/www/html/dev search-replace https://"${SITE_FULL_DOMAIN}" https://dev."${SITE_DOMAIN_EXCLUDING_WWW}" --all-tables

####################################################################################################
#### N. SS-Sync-Development: Reset Permissions (WordPress Packages) ################################
####################################################################################################

source "${PATH_SS_PERMS_WORDPRESS_PACKAGES}"

####################################################################################################
#### O. SS-Sync-Development: End If ################################################################
####################################################################################################

fi

####################################################################################################
#### SlickStack: Reset Permissions (SlickStack Scripts) ############################################
####################################################################################################

## we include this permissions reset in all cron jobs and bash scripts for redundancy ##
## chmod 0700 means only the root/sudo users can execute any SlickStack scripts ##

## THIS SNIPPET DOES NOT RELY ON SS-CONFIG OR SS-FUNCTIONS
## SNIPPET: ss bash scripts, ss cron jobs
## UPDATED: 02JUL2022

chown root:root /var/www/ss* ## must be root:root
chown root:root /var/www/crons/*cron* ## must be root:root
chown root:root /var/www/crons/custom/*cron* ## must be root:root
chmod 0700 /var/www/ss* ## 0700 means only root/sudo can execute
chmod 0700 /var/www/crons/*cron* ## 0700 means only root/sudo can execute
chmod 0700 /var/www/crons/custom/*cron* ## 0700 means only root/sudo can execute
