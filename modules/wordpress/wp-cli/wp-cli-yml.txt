####################################################################################################
#### B. WP-CLI.yml: Disabled Commands ##############################################################
####################################################################################################

## several [wp] commands are disabled below to avoid potential SlickStack conflicts ##
## SlickStack should be used to install databases, WordPress, and WP-CLI ##

disabled_commands:
 - config create
 - config delete
 - config edit
 - config set
 - db cli
 - db create
 - db drop
 - db reset
 - package
 - server
 - shell

#### D. WP-CLI.yml: Environment Aliases ############################################################
####################################################################################################

## these are shortcut aliases for use with [wp] command to specify which environment ##
## you can use any combination of these to run tasks on SlickStack servers ##

@dev:
  path: /var/www/html/dev
@development:
  path: /var/www/html/dev
@staging:
  path: /var/www/html/staging
@prod:
  path: /var/www/html
@production:
  path: /var/www/html
@both:
 - @staging
 - @production
@all:
 - @development
 - @staging
 - @production
